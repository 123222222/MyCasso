// Create display object

// Create RoboEyes object (FluxGarage port for ST7789)

  Serial.println("\nğŸ¤– RoboEyes ST7789 Demo");

  tft.fillScreen(ST77XX_BLACK);
  
  // Initialize RoboEyes (width,height,maxFPS)
  // Landscape: 320x240
  // TÄƒng FPS Ä‘á»ƒ giáº£m cáº£m giÃ¡c nháº¥p nhÃ¡y khi váº½ dirty-rect

  // Configure eyes - tÄƒng gáº¥p Ä‘Ã´i tá»« 40x50 lÃªn 80x100
  eyes.setWidth(80, 80);
  eyes.setHeight(100, 100);
  eyes.setBorderradius(30, 30); // tÄƒng gáº¥p Ä‘Ã´i tá»« 15
  eyes.setSpacebetween(5); // giáº£m xuá»‘ng 5 - 2 máº¯t sÃ¡t nhau

  eyes.setDisplayColors(ST77XX_BLACK, ST77XX_CYAN);

  // Enable auto behaviors (use FluxGarage API names)
  // Auto-blink enabled
  eyes.setAutoblinker(ON, 16, 3); // active, interval (s), variation (s)
  // Auto idle ON - tá»± Ä‘á»™ng Ä‘á»•i hÆ°á»›ng (8 hÆ°á»›ng) + mood (5 loáº¡i) má»—i 4 giÃ¢y

  printHelp();
}

void loop() {
  // Update eyes (handles auto blink and idle)
  eyes.update();
  
  // Check serial commands
  if (Serial.available()) {
    char cmd = Serial.read();
    handleCommand(cmd);
  }
}
